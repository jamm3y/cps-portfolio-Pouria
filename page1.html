This page shows a game I created using JavaScript. JavaScript lets games run directly in a web browser.

Playable Game

Guess a number between 1 and 10:

My JavaScript Game Code


// You can paste or edit your JavaScript game code here

let secretNumber = Math.floor(Math.random() * 10) + 1;

function checkGuess() {
    let guess = document.getElementById("guessInput").value;
    let result = document.getElementById("result");

    if (guess == secretNumber) {
        result.textContent = "Correct! You guessed the number!";
    } else {
        result.textContent = "Wrong guess. Try again!";
    }
}
